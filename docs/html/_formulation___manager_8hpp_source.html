196  forcing_params get_global_forcing_params(std::string identifier) {
197  std::string path = this->global_forcing.at("path").as_string();
198 
199  if (this->global_forcing.count("file_pattern") == 0) {
200  return forcing_params(
201  path,
202  this->global_forcing.at("start_time").as_string(),
203  this->global_forcing.at("end_time").as_string()
204  );
205  }
206 
207  // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
208  if (path.compare(path.size() - 1, 1, "/") != 0) {
209  path += "/";
210  }
211 
212  std::string filepattern = this->global_forcing.at("file_pattern").as_string();
213 
214  int id_index = filepattern.find("{{ID}}");
215 
216  // If an index for '{{ID}}' was found, we can count on that being where the id for this realization can be found.
217  // For instance, if we have a pattern of '.*{{ID}}_14_15.csv' and this is named 'cat-87',
218  // this will match on 'stuff_example_cat-87_14_15.csv'
219  if (id_index != std::string::npos) {
220  filepattern = filepattern.replace(id_index, sizeof("{{ID}}") - 1, identifier);
221  }
222 
223  // Create a regular expression used to identify proper file names
224  std::regex pattern(filepattern);
225 
226  // A stream providing the functions necessary for evaluating a directory:
227  // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
228  DIR *directory = nullptr;
229 
230  // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
231  struct dirent *entry = nullptr;
232 
233  // Attempt to open the directory for evaluation
234  directory = opendir(path.c_str());
235 
236  // If the directory could be found, we can go ahead and iterate
237  if (directory != nullptr) {
238  while ((entry = readdir(directory))) {
239  // If the entry is a regular file or symlink AND the name matches the pattern,
240  // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
241  if ((entry->d_type == DT_REG or entry->d_type == DT_LNK) and std::regex_match(entry->d_name, pattern)) {
242  return forcing_params(
243  path + entry->d_name,
244  this->global_forcing.at("start_time").as_string(),
245  this->global_forcing.at("end_time").as_string()
246  );
247  }
248  }
249  }
250  else {
251  // The directory wasn't found; forcing data cannot be retrieved
252  throw std::runtime_error("No directory for forcing data was found at: " + path);
253  }
254 
255  closedir(directory);
256 
257  throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
258  }
